388 // gpu specific implementations instead, and should not change these
389 // functions.
390 template <typename Dtype>
391 inline Dtype Layer<Dtype>::Forward(const vector<Blob<Dtype>*>& bottom,
392  vector<Blob<Dtype>*>* top) {
393  Dtype loss = 0;
394  switch (Caffe::mode()) {
395  case Caffe::CPU:
396  Forward_cpu(bottom, top);
397  for (int top_id = 0; top_id < top->size(); ++top_id) {
398  if (!this->loss(top_id)) { continue; }
399  const int count = (*top)[top_id]->count();
400  const Dtype* data = (*top)[top_id]->cpu_data();
401  const Dtype* loss_weights = (*top)[top_id]->cpu_diff();
402  loss += caffe_cpu_dot(count, data, loss_weights);
403  }
404  break;
405  case Caffe::GPU:
406  Forward_gpu(bottom, top);
407 #ifndef CPU_ONLY
408  for (int top_id = 0; top_id < top->size(); ++top_id) {
409  if (!this->loss(top_id)) { continue; }
410  const int count = (*top)[top_id]->count();
411  const Dtype* data = (*top)[top_id]->gpu_data();
412  const Dtype* loss_weights = (*top)[top_id]->gpu_diff();
413  Dtype blob_loss = 0;
414  caffe_gpu_dot(count, data, loss_weights, &blob_loss);
415  loss += blob_loss;
416  }
417 #endif
418  break;
419  default:
420  LOG(FATAL) << "Unknown caffe mode.";
421  }
422  return loss;
423 }
